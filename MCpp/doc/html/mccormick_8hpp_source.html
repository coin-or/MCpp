02335 ( const double xL, const double xU, puniv f, const double*rusr,
02336   const int*iusr )

02339   const double fL = f(xL,rusr,iusr), fU = f(xU,rusr,iusr);

02345 template <typename T> inline double
02346 McCormick<T>::_goldsect_iter
02347 ( const bool init, const double a, const double fa, const double b,
02348   const double fb, const double c, const double fc, puniv f,
02349   const double*rusr, const int*iusr )
02350 // a and c are the current bounds; the minimum is between them.
02351 // b is a center point
02352 {
02353   static unsigned int iter;
02354   iter = ( init? 1: iter+1 );
02355   const double phi = 2.-(1.+std::sqrt(5.))/2.;
02356   bool b_then_x = ( c-b > b-a );
02357   double x = ( b_then_x? b+phi*(c-b): b-phi*(b-a) );
02358   if( std::fabs(c-a) < options.ENVEL_TOL*(std::fabs(b)+std::fabs(x)) 
02359    || iter > options.ENVEL_MAXIT ) return (c+a)/2.;
02360   double fx = f(x,rusr,iusr);
02361   if( b_then_x )
02362     return( fa*fx<0? _goldsect_iter( false, a, fa, b, fb, x, fx, f, rusr, iusr ):
02363                       _goldsect_iter( false, b, fb, x, fx, c, fc, f, rusr, iusr ) );
02364   return( fa*fb<0? _goldsect_iter( false, a, fa, x, fx, b, fb, f, rusr, iusr ):
02365                     _goldsect_iter( false, x, fx, b, fb, c, fc, f, rusr, iusr ) );
02366 }
