00559 template <class T> inline T
00560 Specbnd<T>::_LambdaS
00561 ( const fadbad::F<T> &a, unsigned int n )
00562 {
00563   if( !n ) return 0.;
00564   if( n == 1 ) return a.size()? Op<T>::sqr( a[0] ): 0.;
00565   double upbnd=0.;
00566   for (unsigned int i=0; i<n; i++)
00567     if( a.size() ) upbnd += Op<T>::u( Op<T>::sqr( a[i] ) );
00568   return Op<T>::zeroone() * upbnd;
00569 }

00571 template <class T> inline T
00572 Specbnd<T>::_LambdaT
00573 ( const fadbad::F<T> &a, const fadbad::F<T> &b, unsigned int n )
00574 {
00575   if( !n ) return 0.;
00576   if( n == 1 ) return (a.size() && b.size())? 2.*a[0]*b[0]: 0.;
00577   double upbnda=0., upbndb=0.;
00578   for (unsigned int i=0; i<n; i++){
00579     if( a.size() ) upbnda += Op<T>::u( Op<T>::sqr( a[i] ) );
00580     if( b.size() ) upbndb += Op<T>::u( Op<T>::sqr( b[i] ) );
00581   }
00582   T lamb = 2.*(Op<T>::zeroone()-0.5) * std::sqrt(upbnda*upbndb);
00583   for (unsigned int i=0; i<n; i++)
00584     if( a.size() && b.size() ) lamb += a[i]*b[i];
00585   return lamb;
00586 }

00589 Specbnd<T>::spectrum
00590 ( const fadbad::B< fadbad::F< double > >* D2X )
00591 {
00592   if( !D2X ) throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::SPECTR );
00593   const unsigned int N = D2X->val().size();
00594   if( !N ) throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::SPECTR );
00595   double*H = new double[N*N];
00596 
00597   for( unsigned int j=0; j<N; j++ )
00598     for( unsigned int i=j; i<N; i++ )
00599       H[j*N+i] = H[i*N+j] = ( i==j? D2X[i].deriv(0).deriv(i):
00600         0.5*(D2X[i].deriv(0).deriv(j)+D2X[j].deriv(0).deriv(i)) );
00601 #ifdef MC__SPECBND_DEBUG_SPECTRUM
00602   mc::display( N, N, H, N, "\nMatrix H", std::cout );
00603 #endif
00604 
00605   double*D = mc::dsyev_wrapper( N, H );
00606   delete[] H;
00607   if( !D ) throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::SPECTR );
00608   std::pair<double,double> spbnd = std::make_pair( min(N,D), max(N,D) );
00609   delete[] D;
00610   return spbnd;

00650 template <typename T> inline std::pair<double,double>
00651 Specbnd<T>::_gershgorin_bound
00652 ( const fadbad::B< fadbad::F< T > >* D2X )
00653 {
00654   const unsigned int N = D2X->val().size();
00655   if( !N ) throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::HESSBND );
00656 
00657   std::pair<double,double> spbnd;
00658   for( unsigned int i=0; i<N; i++ ){
00659     double ri = 0.;
00660     for( unsigned int j=0; j<N; j++ ){
00661       if( j == i ) continue;
00662       T D2Fij;
00663       if( !inter( D2Fij, D2X[i].deriv(0).deriv(j), D2X[i].deriv(0).deriv(j) ) )
00664         D2Fij = D2X[i].deriv(0).deriv(j); 
00665       ri += Op<T>::abs( D2Fij );
00666     }
00667 
00668     if( !i ){
00669       spbnd.first  = Op<T>::l(D2X[i].deriv(0).deriv(i)) - ri;
00670       spbnd.second = Op<T>::u(D2X[i].deriv(0).deriv(i)) + ri;
00671     }
00672     else{
00673       spbnd.first  = std::min( spbnd.first,  Op<T>::l(D2X[i].deriv(0).deriv(i)) - ri );
00674       spbnd.second = std::max( spbnd.second, Op<T>::u(D2X[i].deriv(0).deriv(i)) + ri );
00675     }
00676   }
00677   return spbnd;
00678 }

00680 template <typename T> inline std::pair<double,double>
00681 Specbnd<T>::_hertzrohn_bound
00682 ( const fadbad::B< fadbad::F< T > >* D2X )
00683 {
00684   const unsigned int N = D2X->val().size();
00685   if( !N ) throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::HESSBND );
00686 
00687   // Form matrix S^{(N)} recursively
00688   unsigned int col_S = 2;
00689   short *S = new short[col_S];
00690   S[0] = 1; S[1] = -1;
00691 #ifdef MC__SPECBND_DEBUG_HESSBND
00692   mc::display( 1, col_S, S, 1, "\nMatrix S1", std::cout );
00693 #endif
00694   for( unsigned int k=1; k<N; k++, col_S*=2 ){
00695     short *Sprev = new short[k*col_S];
00696     for( unsigned int i=0; i<k; i++ )
00697       for( unsigned int j=0; j<col_S; j++ )
00698         Sprev[j*k+i] = S[j*k+i];
00699     delete[] S; S = new short[(k+1)*2*col_S];
00700     for( unsigned int i=0; i<k; i++ ){
00701       for( unsigned int j=0; j<col_S; j++ )
00702         S[j*(k+1)+i] = S[(col_S+j)*(k+1)+i] = Sprev[j*k+i];
00703     }
00704     for( unsigned int j=0; j<col_S; j++ ){
00705       S[j*(k+1)+k] = 1; S[(col_S+j)*(k+1)+k] = -1;
00706     }
00707     delete[] Sprev;
00708 #ifdef MC__SPECBND_DEBUG_HESSBND
00709     mc::display( k+1, 2*col_S, S, k+1, "\nMatrix Sk", std::cout );
00710 #endif
00711   }
00712 
00713   // Compute lower and upper bound on spectral radius
00714   std::pair<double,double> spbnd;
00715   double *Lk = new double[N*N], *Uk = new double[N*N];
00716   for( unsigned int k=0; k<col_S; k++ ){
00717     
00718     for( unsigned int j=0; j<N; j++ )
00719       for( unsigned int i=j; i<N; i++ ){
00720         if( i == j ){
00721      Lk[i*N+i] = Op<T>::l( D2X[i].deriv(0).deriv(i) );
00722      Uk[i*N+i] = Op<T>::u( D2X[i].deriv(0).deriv(i) );
00723           continue;
00724    }
00725         T D2Fij;
00726         if( !inter( D2Fij, D2X[i].deriv(0).deriv(j), D2X[j].deriv(0).deriv(i) ) )
00727           D2Fij = D2X[i].deriv(0).deriv(j);
00728         Lk[i*N+j] = Lk[j*N+i] = S[k*N+i]*S[k*N+j]==1? Op<T>::l( D2Fij ): Op<T>::u( D2Fij );
00729         Uk[i*N+j] = Uk[j*N+i] = S[k*N+i]*S[k*N+j]==1? Op<T>::u( D2Fij ): Op<T>::l( D2Fij );
00730       }
00731 #ifdef MC__SPECBND_DEBUG_HESSBND
00732     mc::display( N, N, Lk, N, "\nMatrix Lk", std::cout );
00733     mc::display( N, N, Uk, N, "\nMatrix Uk", std::cout );
00734 #endif
00735 
00736     double*DLk = mc::dsyev_wrapper( N, Lk );
00737     if( !DLk ){
00738       delete[] Lk; delete[] Uk; delete[] S;
00739       throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::HESSBND );
00740     }
00741 #ifdef MC__SPECBND_DEBUG_HESSBND
00742     mc::display( 1, N, DLk, 1, "\nMatrix DLk", std::cout );
00743 #endif
00744     spbnd.first = k? std::min( spbnd.first, min(N,DLk) ): min(N,DLk);
00745     delete[] DLk;
00746 
00747     double*DUk = mc::dsyev_wrapper( N, Uk );
00748     if( !DUk ){
00749       delete[] Lk; delete[] Uk; delete[] S;
00750       throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::HESSBND );
00751     }
00752 #ifdef MC__SPECBND_DEBUG_HESSBND
00753     mc::display( 1, N, DUk, 1, "\nMatrix DUk", std::cout );
00754 #endif
00755     spbnd.second = k? std::max( spbnd.second, max(N,DUk) ): max(N,DUk);
00756     delete[] DUk;
00757 
00758   }
00759   delete[] Lk; delete[] Uk; delete[] S;
00760   return spbnd;
00761 }

00832     delete[] Sprev;
00833 #ifdef MC__SPECBND_DEBUG_HESSBND
00834     mc::display( k+1, 2*col_S, S, k+1, "\nMatrix Sk", std::cout );
00835 #endif
00836   }
00837 
00838 #ifdef MC__SPECBND_DEBUG_HESSBND
00839   T *H = new T[N*N];
00840   for( unsigned int j=0; j<N; j++ )
00841     for( unsigned int i=j; i<N; i++ ){
00842       if( i == j ){
00843         H[i*N+i] = D2F.deriv(i).deriv(i);
00844         continue;
00845       }
00846       T D2Fij;
00847       if( !inter( D2Fij, D2F.deriv(i).deriv(j), D2F.deriv(j).deriv(i) ) )
00848         D2Fij = D2F.deriv(i).deriv(j);
00849       H[i*N+j] = H[j*N+i] = D2Fij;
00850     }
00851   mc::display( N, N, H, N, "\nMatrix H", std::cout );
00852   delete[] H;
00853 #endif
00854 
00855   // Compute lower and upper bound on spectral radius
00856   std::pair<double,double> spbnd;
00857   double *Lk = new double[N*N], *Uk = new double[N*N];
00858   for( unsigned int k=0; k<col_S; k++ ){
00859     
00860     for( unsigned int j=0; j<N; j++ )
00861       for( unsigned int i=j; i<N; i++ ){
00862         if( i == j ){
00863      Lk[i*N+i] = Op<T>::l( D2F.deriv(i).deriv(i) );
00864      Uk[i*N+i] = Op<T>::u( D2F.deriv(i).deriv(i) );
00865           continue;
00866    }
00867         T D2Fij;
00868         if( !inter( D2Fij, D2F.deriv(i).deriv(j), D2F.deriv(j).deriv(i) ) )
00869           D2Fij = D2F.deriv(i).deriv(j);
00870         Lk[i*N+j] = Lk[j*N+i] = S[k*N+i]*S[k*N+j]==1? Op<T>::l( D2Fij ): Op<T>::u( D2Fij );
00871         Uk[i*N+j] = Uk[j*N+i] = S[k*N+i]*S[k*N+j]==1? Op<T>::u( D2Fij ): Op<T>::l( D2Fij );
00872       }
00873 #ifdef MC__SPECBND_DEBUG_HESSBND
00874     mc::display( N, N, Lk, N, "\nMatrix Lk", std::cout );
00875     mc::display( N, N, Uk, N, "\nMatrix Uk", std::cout );
00876 #endif
00877 
00878     double*DLk = mc::dsyev_wrapper( N, Lk );
00879     if( !DLk ){
00880       delete[] Lk; delete[] Uk; delete[] S;
00881       throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::HESSBND );
00882     }
00883 #ifdef MC__SPECBND_DEBUG_HESSBND
00884     mc::display( 1, N, DLk, 1, "\nMatrix DLk", std::cout );
00885 #endif
00886     spbnd.first = k? std::min( spbnd.first, min(N,DLk) ): min(N,DLk);
00887     delete[] DLk;
00888 
00889     double*DUk = mc::dsyev_wrapper( N, Uk );
00890     if( !DUk ){
00891       delete[] Lk; delete[] Uk; delete[] S;
00892       throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::HESSBND );
00893     }
00894 #ifdef MC__SPECBND_DEBUG_HESSBND
00895     mc::display( 1, N, DUk, 1, "\nMatrix DUk", std::cout );
00896 #endif
00897     spbnd.second = k? std::max( spbnd.second, max(N,DUk) ): max(N,DUk);
00898     delete[] DUk;
00899 
00900   }
00901   delete[] Lk; delete[] Uk; delete[] S;
00902   return spbnd;
00903 }
