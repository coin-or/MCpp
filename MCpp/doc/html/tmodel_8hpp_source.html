01522 template <typename T> inline TVar<T>
01523 TModel<T>::_univ_bernstein
01524 ( const TVar<T>&TV, puniv f, puniv df, punivext If,
01525   punivext d2If, const double*rusr, const int*iusr )
01526 {
01527   assert( TV._TM == this );
01528   
01529   const T B( TV.B() );
01530   const TVar<T> TVs( (TV-Op<T>::l(B))/Op<T>::diam(B) );
01531   TVar<T> TV2( this, 0. ), MON( 1. );
01532   for( unsigned int j=0; j<=_nord; j++ ){
01533     double sign = ( j%2? -1.: 1. );
01534     _cbern[j] = sign * _get_binom(_nord,j) * f( Op<T>::l(B), rusr, iusr );
01535     for( unsigned int i=1; i<=j; i++ ){
01536       sign *= -1.;
01537       _cbern[j] += sign * _get_binom(_nord,i) * _get_binom(_nord-i,j-i)
01538         * f( Op<T>::l(B)+(double)i/(double)_nord*Op<T>::diam(B), rusr, iusr );
01539     }
01540     TV2 += _cbern[j] * MON;
01541     MON *= TVs;
01542   }
01543 
01544   T R( - d2If( B, rusr, iusr ) * sqr(Op<T>::diam(B)) / 2. / _nord );
01545   if( Op<T>::l( R ) * Op<T>::u( R ) < 0 )
01546     return TV2 + Op<T>::zeroone() * R;
01547 
01548   if( !options.BERNSTEIN_OPT ){
01549     T R0( If(B,rusr,iusr) );
01550     R = Op<T>::zeroone() * ( Op<T>::u(R) <= 0.?
01551          -std::min( Op<T>::diam(R0), -Op<T>::l(R) ):
01552           std::min( Op<T>::diam(R0),  Op<T>::u(R) ) );
01553     return TV2 + R;
01554   }
01555 
01556   std::pair<unsigned int,const double*> bern = std::make_pair(_nord+1,_cbern);
01557   double xopt = _goldsect( 0., 1., _dgap_bernstein,  rusr, iusr, df, B, bern );

01611   const double phi = 2.-(1.+std::sqrt(5.))/2.;
01612   bool b_then_x = ( c-b > b-a );

01614   if( std::fabs(c-a) < options.BERNSTEIN_TOL*(std::fabs(b)+std::fabs(x)) 
01615    || iter > options.BERNSTEIN_MAXIT ) return (c+a)/2.;
01616   double fx = fopt( x, rusr, iusr, df, Ix, bern );

01618     return( fa*fx<0?
01619       _goldsect_iter( false, a, fa, b, fb, x, fx, fopt, rusr, iusr, df, Ix, bern ):
01620      _goldsect_iter( false, b, fb, x, fx, c, fc, fopt, rusr, iusr, df, Ix, bern ) );

01622      _goldsect_iter( false, a, fa, x, fx, b, fb, fopt, rusr, iusr, df, Ix, bern ):
